import { Component } from '@angular/core';
import { Field } from '../../../core/field/field';

/** Holds the templates to output item field values. */
@Component({
  selector: 'ec-default-output',
  templateUrl: './default-output.component.html',
})
export class DefaultOutputComponent {
  /** The field that should be displayed */
  field: Field<any>;
}

</div>
<div [ngSwitch]="field.view" *ngIf="item&&item?.display(field.property)">

  <ul class="labels-list" *ngSwitchCase="'labels'">

      <div class="label">
        {{label}}

  </ul>

    <i class="ec-icon check-box" *ngIf="item?.resolve(field.property)"></i>
  </div>

  </div>
  <div class="ec-list-avatar" *ngSwitchCase="'avatars'">
    <img [src]="url" *ngFor="let url of item?.display(field.property)">
  </div>

    {{item?.display(field.property)}}
  </span>
</div>
