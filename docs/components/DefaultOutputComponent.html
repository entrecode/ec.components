import { Component } from '@angular/core';
import { Field } from '@ec.components/core/src/field/field';
import { Item } from '@ec.components/core/src/item/item';

/** Holds the templates to output item field values. */
@Component({
  selector: 'ec-default-output',
  styleUrls: ['./default-output.component.scss'],
  templateUrl: './default-output.component.html',
})
export class DefaultOutputComponent {
  /** The field that should be displayed */
  field: Field;
  /** The item that is targeted by the input */
  item: Item<any>;
}

<div [ngSwitch]="field.view" *ngIf="item">
  <ul class="ec-output_tags-list" *ngSwitchCase="'tags'">
    <li *ngFor="let label of item?.display(field.property) | maxItems:field.maxItems||10">
      <div class="tag">
        {{label}}
      </div>
    </li>
    <li class="tag" *ngIf="item?.display(field.property)?.length>(field.maxItems||10)">
      ...
    </li>
  </ul>

  <div class="ec-output_tag" *ngSwitchCase="'tag'">
    <span class="tag" *ngIf="item?.display(field.property)">{{item?.display(field.property)}}</span>
  </div>

  <div *ngSwitchCase="'boolean'" class="ec-output_boolean" [class.is-true]="item?.resolve(field.property)">
    <span>&#10004;</span>
  </div>
  <div *ngSwitchCase="'date'" class="ec-output_date" [title]="item?.display(field.property)">
    {{item?.display(field.property)}}
  </div>

  <div class="ec-output_mail" *ngSwitchCase="'email'">
    <a [href]="'mailto:'+item?.display(field.property)">{{item?.display(field.property)}}</a>
  </div>
  <div class="ec-output_url" *ngSwitchCase="'url'">
    <a [href]="item?.display(field.property)">{{item?.display(field.property)}}</a>
  </div>

    <span class="ec-output_color-sample" [style.backgroundColor]="item?.display(field.property)"></span>
    <span class="ec-output_color-property">{{item?.display(field.property)}}</span>
  </div>

  <div class="ec-output_asset" *ngSwitchCase="'asset'" [class.is-true]="item?.resolve(field.property)">
    <ec-icon name="img"></ec-icon>

    <img [src]="item?.display(field.property)">
  </div>

  <div class="ec-output_preview" *ngSwitchCase="'preview'">
    <img [src]="item?.display(field.property)">
  </div>

  <a *ngSwitchCase="'link'" (click)="item?.action(field.property, $event)" [ngClass]="field.class">
    {{item?.display(field.property)}}
    <ec-icon *ngIf="field.icon" [name]="field.icon"></ec-icon>
  </a>
  <div *ngSwitchCase="'textarea'" class="ec-output_textarea">
    <!--{{item?.display(field.property)}}-->

  </div>
